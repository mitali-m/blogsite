sudo apt update
sudo apt upgrade
sudo apt autoremove

sudo apt install tomcat9
sudo nano /etc/default/tomcat9 --> add AUTHBIND=yes at the end of the file

sudo apt install authbind
ls -al /etc/authbind/byport/
sudo touch /etc/authbind/byport/80
sudo chmod 500 /etc/authbind/byport/80
sudo chown tomcat /etc/authbind/byport/80

sudo service tomcat9 status
sudo service tomcat9 stop

sudo nano /var/lib/tomcat9/conf/server.xml  --> change the connector port from 8080 to 80
sudo service tomcat9 start

curl localhost
